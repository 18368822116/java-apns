42  /***
43   * Represents a builder for constructing Payload requests, as
44   * specified by Apple Push Notification Programming Guide.
45   */
46  public final class PayloadBuilder {
47      private static final ObjectMapper mapper = new ObjectMapper();

49      private final Map<String, Object> root;

86      /***
87       * Sets the notification badge to be displayed next to the

97      public PayloadBuilder badge(int badge) {

99          return this;

102     /***
103      * Requests clearing of the badge number next to the application

108      * @return this
109      */
110     public PayloadBuilder clearBadge() {
111         return badge(0);
112     }

114     /***
115      * Sets the value of action button (the right button to be
116      * displayed).  The default value is "View".
117      *
118      * The value can be either the simple String to be displayed or

128     public PayloadBuilder actionKey(String actionKey) {
129         customAlert.put("action-loc-key", actionKey);
130         return this;
131     }

136      * This is an alias to {@code actionKey(null)}

199     public int length() {
200         String str = this.copy().toString();
201         int payloadLength = Utilities.toUTF8Bytes(str).length;
202         return payloadLength;
203     }
204 
205     public boolean isTooLong() {
206         return this.length() > Utilities.MAX_PAYLOAD_LENGTH;
207     }
208 
209     public PayloadBuilder resizeAlertBody(int payloadLength) {
210         int currLength = length();
211         if (currLength < payloadLength)
212             return this;
213 
214         int d = currLength - payloadLength;
215         String body = (String)aps.get("alert");
216 
217         if (body.length() < d)
218             aps.remove("alert");
219         else
220             aps.put("alert", body.subSequence(0, body.length() - d));
221 
222         return this;
223     }
224 
225     public PayloadBuilder shrinkBody() {
226         return resizeAlertBody(Utilities.MAX_PAYLOAD_LENGTH);
227     }

229     /***
230      * Returns the JSON String representation of the payload
231      * according to Apple APNS specification
232      *
233      * @return  the String representation as expected by Apple
234      */
235     public String build() {
236         if (!(customAlert.isEmpty()
237                 || customAlert.equals(aps.get("alert")))) {
238             if (aps.containsKey("alert")) {
239                 String alertBody = (String)aps.get("alert");
240                 customAlert.put("body", alertBody);
241             }
242             aps.put("alert", customAlert);
243         }
244         root.put("aps", aps);
245         try {
246             return mapper.writeValueAsString(root);
247         } catch (Exception e) {
248             throw new RuntimeException(e);
249         }
250     }
251 
252     @Override
253     public String toString() {
254         return this.build();
255     }

257     private PayloadBuilder(Map<String, Object> root, Map<String, Object> aps, Map<String, Object> customAlert) {
258         this.root = new HashMap<String, Object>(root);
259         this.aps = new HashMap<String, Object>(aps);
260         this.customAlert = new HashMap<String, Object>(customAlert);
261     }
262 
263     public PayloadBuilder copy() {
264         return new PayloadBuilder(this.root, this.aps, this.customAlert);
265     }
