59      public ApnsConnectionImpl(SocketFactory factory, String host, int port, ReconnectPolicy reconnectPolicy, ApnsDelegate delegate) {
60          this.factory = factory;
61          this.host = host;
62          this.port = port;
63          this.reconnectPolicy = reconnectPolicy;
64          this.delegate = delegate == null ? ApnsDelegate.EMPTY : delegate;
65      }

67      public synchronized void close() {
68          try {
69              if (socket != null)
70                  this.socket.close();
71          } catch (IOException e) {
72              logger.debug("Error while closing socket connection", e);
73          }
74      }

77      // This method is only called from sendMessage.  sendMessage
78      // has the required logic for retrying
79      private Socket socket;
80      private synchronized Socket socket() throws NetworkIOException {
81          if (reconnectPolicy.shouldReconnect()) {
82              Utilities.close(socket);
83              socket = null;
84          }
85  
86          if (socket == null || socket.isClosed()) {
87              try {
88                  socket = factory.createSocket(host, port);
89                  reconnectPolicy.reconnected();
90                  logger.debug("Made a new connection to APNS");
91              } catch (IOException e) {
92                  logger.error("Couldn't connect to APNS server", e);
93                  throw new NetworkIOException(e);
94              }
95          }
96          return socket;
97      }

99      int DELAY_IN_MS = 1000;
100 
101     private static final int RETRIES = 3;
102     public synchronized void sendMessage(ApnsNotification m) throws NetworkIOException {
103         int attempts = 0;
104         while (true) {
105             try {
106                 attempts++;
107                 Socket socket = socket();
108                 socket.getOutputStream().write(m.marshall());
109                 socket.getOutputStream().flush();
110                 delegate.messageSent(m);
111 
112                 logger.debug("Message \"{}\" sent", m);
113 
114                 attempts = 0;
115                 break;
116             } catch (Exception e) {
117                 if (attempts >= RETRIES) {
118                     logger.error("Couldn't send message " + m, e);
119                     delegate.messageSendFailed(m, e);
120                     Utilities.wrapAndThrowAsRuntimeException(e);
121                 }
122                 logger.warn("Failed to send message " + m + "... trying again", e);
123                 // The first failure might be due to closed connection
124                 // don't delay quite yet
125                 if (attempts != 1)
126                     Utilities.sleep(DELAY_IN_MS);
127                 Utilities.close(socket);
128                 socket = null;
129             }
130         }
131     }
