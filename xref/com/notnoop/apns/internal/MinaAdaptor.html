34      private NioSocketConnector createNioSocketConnector(SSLContext sslContext) {
35          NioSocketConnector connector = new NioSocketConnector();
36          connector.setConnectTimeoutMillis(30 * 1000L);
37          connector.setHandler(new IoHandlerAdapter());
38          SslFilter sslFilter = new SslFilter(sslContext);
39          sslFilter.setUseClientMode(true);
40          connector.getFilterChain().addLast("SSL", sslFilter);
41          return connector;
