31  package com.notnoop.apns.internal;
32  
33  import java.io.IOException;
34  import java.net.Socket;
35  
36  import javax.net.SocketFactory;
37  
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  
41  import com.notnoop.apns.ApnsNotification;
42  
43  public class ApnsConnection {
44      private static final Logger logger = LoggerFactory.getLogger(ApnsConnection.class);
45  
46      private final SocketFactory factory;
47      private final String host;
48      private final int port;
49  
50      public ApnsConnection(SocketFactory factory, String host, int port) {
51          this.factory = factory;
52          this.host = host;
53          this.port = port;
54      }
55  
56      protected void close() throws IOException {
57          this.socket.close();
58      }
59  
60      private Socket socket;
61      private Socket socket() {
62          while (socket == null || socket.isClosed()) {
63              try {
64                  socket = factory.createSocket(host, port);
65                  logger.debug("Made a new connection to APNS");
66              } catch (Exception e) {
67                  logger.error("Couldnt' connec to APNS server", e);
68              }
69          }
70          return socket;
71      }
72  
73      int DELAY_IN_MS = 1000;
74  
75      private static final int RETRIES = 3;
76      protected synchronized void sendMessage(ApnsNotification m) {
77          int attempts = 0;
78          while (true) {
79              try {
80                  attempts++;
81                  Socket socket = socket();
82                  socket.getOutputStream().write(m.marshall());
83                  socket.getOutputStream().flush();
84                  logger.debug("Message \"{}\" sent", m);
85  
86                  attempts = 0;
87                  break;
88              } catch (Exception e) {
89                  if (attempts >= RETRIES) {
90                      logger.error("Couldn't send message " + m, e);
91                      throw new RuntimeException(e);
92                  }
93                  logger.warn("Failed to send message " + m + "... trying again", e);
94                  Utilities.sleep(DELAY_IN_MS);
95                  Utilities.close(socket);
96                  socket = null;
97              }
98          }
99      }
