83      public static SSLContext newSSLContext(InputStream cert, String password,
84           String ksType, String ksAlgorithm) throws InvalidSSLConfig {
85          try {
86              KeyStore ks = KeyStore.getInstance(ksType);
87              ks.load(cert, password.toCharArray());
88  
89              // Get a KeyManager and initialize it
90              KeyManagerFactory kmf = KeyManagerFactory.getInstance(ksAlgorithm);
91              kmf.init(ks, password.toCharArray());
92  
93              // Get a TrustManagerFactory and init with KeyStore
94              TrustManagerFactory tmf = TrustManagerFactory.getInstance(ksAlgorithm);
95              tmf.init(ks);
96  
97              // Get the SSLContext to help create SSLSocketFactory
98              SSLContext sslc = SSLContext.getInstance("TLS");
99              sslc.init(kmf.getKeyManagers(), null, null);
100             return sslc;
101         } catch (Exception e) {
102             throw new InvalidSSLConfig(e);
103         }
104     }

106     private static final Pattern pattern = Pattern.compile("[ -]");
107     public static byte[] decodeHex(String deviceToken) {
108      String hex = pattern.matcher(deviceToken).replaceAll("");
109 
110      byte[] bts = new byte[hex.length() / 2];
111      for (int i = 0; i < bts.length; i++) {
112          bts[i] = (byte) (charval(hex.charAt(2*i)) * 16 + charval(hex.charAt(2*i + 1)));
113      }
114      return bts;
115     }

117     private static int charval(char a) {
118         if ('0' <= a && a <= '9')
119             return (a - '0');
120         else if ('a' <= a && a <= 'f')
121             return (a - 'a') + 10;
122         else if ('A' <= a && a <= 'F')
123             return (a - 'A') + 10;
124         else
125             throw new RuntimeException("Invalid hex character: " + a);
126     }

130     public static String encodeHex(byte[] bytes) {
131      char[] chars = new char[bytes.length * 2];
132 
133      for (int i = 0; i < bytes.length; ++i) {
134          int b = ((int)bytes[i]) & 0xFF;
135          chars[2 * i] = base[b >>> 4];
136          chars[2 * i + 1] = base[b & 0xF];
137      }
138 
139      return new String(chars);
140     }

150     public static byte[] marshall(byte command, byte[] deviceToken, byte[] payload) {
151      ByteArrayOutputStream boas = new ByteArrayOutputStream();
152      DataOutputStream dos = new DataOutputStream(boas);
153 
154      try {
155          dos.writeByte(command);
156          dos.writeShort(deviceToken.length);
157          dos.write(deviceToken);
158          dos.writeShort(payload.length);
159          dos.write(payload);
160          return boas.toByteArray();
161      } catch (IOException e) {
162          throw new AssertionError();
163      }
164     }

166     public static Map<byte[], Integer> parseFeedbackStreamRaw(InputStream in) {
167      Map<byte[], Integer> result = new HashMap<byte[], Integer>();
168 
169      DataInputStream data = new DataInputStream(in);
170 
171      while (true) {
172          try {
173           int time = data.readInt();
174           int dtLength = data.readUnsignedShort();
175           byte[] deviceToken = new byte[dtLength];
176           data.readFully(deviceToken);
177 
178           result.put(deviceToken, time);
179          } catch (EOFException e) {
180           break;
181          } catch (IOException e) {
182           throw new RuntimeException(e);
183          }
184      }
185 
186      return result;
187     }

189     public static Map<String, Date> parseFeedbackStream(InputStream in) {
190      Map<String, Date> result = new HashMap<String, Date>();
191 
192      Map<byte[], Integer> raw = parseFeedbackStreamRaw(in);
193      for (Map.Entry<byte[], Integer> entry : raw.entrySet()) {
194          byte[] dtArray = entry.getKey();
195          int time = entry.getValue(); // in seconds
196 
197          Date date = new Date(time * 1000L);    // in ms
198          String dtString = encodeHex(dtArray);
199          result.put(dtString, date);
200      }
201 
202      return result;
203     }

237     public static byte[] copyOfRange(byte[] original, int from, int to) {
238         int newLength = to - from;
239         if (newLength < 0)
240             throw new IllegalArgumentException(from + " > " + to);
241         byte[] copy = new byte[newLength];
242         System.arraycopy(original, from, copy, 0,
243                          Math.min(original.length - from, newLength));
244         return copy;
245     }

247     public static void wrapAndThrowAsRuntimeException(Exception e) throws NetworkIOException {
248         if (e instanceof IOException)
